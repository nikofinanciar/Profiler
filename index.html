Python Stock Analyzer


import pandas as pd
import numpy as np

def analyze_stock_data(filepath):
    """
    Loads stock data from a CSV file and calculates
    daily returns and rolling volatility.
    """
    try:
        data = pd.read_csv(filepath, index_col='Date', parse_dates=True)
        
        # Calculate daily returns
        data['Daily_Return'] = data['Adj Close'].pct_change()
        
        # Calculate 30-day rolling volatility
        data['Volatility'] = data['Daily_Return'].rolling(window=30).std() * np.sqrt(252)
        
        return data[['Adj Close', 'Daily_Return', 'Volatility']].dropna()
        
    except FileNotFoundError:
        return "Error: Data file not found."
